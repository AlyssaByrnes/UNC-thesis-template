

% Use new proper margins.
\geometry{letterpaper,left=1in,top=1in,right=1in,bottom=1in,nohead}


% double-space text
\doublespacing

% Use new format for headings.  (Differs from Bjoern's template due to changes
% imposed by the graduate school.)

%Put "CHAPTER X: TITLE IN ALL CAPS" on the title line.  \singlespacing allows
% a long title to be broken up if necessary.
% Beware that this calls "\uppercase" on the chapter title.  \MakeUppercase
% is theoretically more proper, but chokes on the "\footnotemark" command.
% "\uppercase" also applies itself to arguments of macros, so be careful.
\titleformat{\chapter}[block]{\singlespacing\fillast\bfseries}{\centering \MakeUppercase{\chaptertitlename} \thechapter:~}{0em}{\uppercase}

% The graduate school got mad at me for having headings bigger than the normal
% text of the dissertation, so I used these commands to force the new format.
\titleformat*{\section}{\bfseries}
\titleformat*{\subsection}{\bfseries}

% Extend to 2in top margins
% Leave 11pts = font size after heading
% These numbers seem random, but they create the proper margins.  I started
% with Bjoern's equally magic-looking template.
\titlespacing{\chapter}{0in}{.62in}{11pt}

% Indent paragraphs four spaces throughout the thesis/dissertation.
\setlength{\parindent}{4ex}

% Tweak spacing of paragraph labels.
\titlespacing{\paragraph}{0in}{0.08in}{0.07in}

% We want numbered subsubsections
\setcounter{secnumdepth}{3}
\setcounter{tocdepth}{3}

% We need to double-space between footnotes.
\setlength{\footnotesep}{13pt}

% We don't want crazy vertical spacing.
\raggedbottom

% We don't want abandoned words.
\clubpenalty=10000 
\widowpenalty=10000

% Prevent awkward hyphenations.
\hyphenation{Raj-kumar}

% Allow long-running math to be broken up.
\allowdisplaybreaks

% Create the proper line before footnotes.  The graduate school got stricter
% about this than when Bjoern wrote his dissertation.
\renewcommand\footnoterule{\noindent \rule{1.5in}{1pt}}

%added by Bryan to prevent footnotes from running across two pages. (dunno if this is grad-school legal yet.)
\interfootnotelinepenalty=10000

\newlength\schedgraphwidth
\setlength\schedgraphwidth{0.7\textwidth}

% Be more flexible in the ratio of float-to-page
 \renewcommand{\dblfloatpagefraction}{.999}
